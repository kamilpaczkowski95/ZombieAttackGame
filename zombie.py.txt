def terminate():
    pygame.quit()

def waitForPlayerToPressKey():
    while True:
        for event in pygame.event.get():
            if event.type == QUIT:
                terminate()
            if event.type == KEYDOWN:
                if event.key == K_ESCAPE: # Escape=wyjście
                    terminate()
                if event.key == K_RETURN:
                    return
